##linux

mkdir -p out/step1
${magma_dir}/magma \
    --annotate window=10,10 \
    --snp-loc ${magma_dir}/g1000_eur.bim \
    --gene-loc ${magma_dir}/NCBI37.3.gene.loc \
    --out out/step1

mkdir -p out/step2
${magma_dir}/magma \
    --bfile ${magma_dir}/g1000_eur \
    --pval ${cancer}cancer_all.tsv use=2,13 ncol=6 \
    --gene-annot out/step1.genes.annot \
    --out out/step2

cancer$NPARAM<-NULL
cancer$N<-NULL

##基因id转换
library(clusterProfiler)
s2ens = bitr(cancer$GENE, fromType="ENTREZID", toType="SYMBOL", OrgDb="org.Hs.eg.db")
data<-subset(cancer,cancer$GENE%in%s2ens$ENTREZID)
data$gene_symbol<-s2ens$SYMBOL
data$GENE<-NULL
data$z_score<-data$ZSTAT
data$ZSTAT<-NULL
write.table(data,file = 'cancer.tsv',row.names = F,sep="\t",quote=F)
